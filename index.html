Welcome ...

I'm a postdoctoral researcher specializing in experimental high-energy physics (HEP), with over a decade of experience working on the CMS experiment at the Large Hadron Collider (LHC). Throughout my career, I've utilized modern computing tools to enhance my research efforts. Having worked at international research institutions, I bring a wealth of diverse experiences to my work.

My expertise also encompasses a significant focus on computing for data processing and analysis, with a specialization in developing algorithms and software in areas such as machine learning, high-performance computing, real-time computing, and Monte Carlo simulation.

I generally enjoy diving into new technical areas, solving complex problems, while continiously learning skills.

template<class T> void ComputeTmpl(
    OpKernelContext* context, 
    unsigned int input_index, 
    const std::vector<unsigned int>& diced_rates,
    unsigned int sum_rates
) const
{
    const Tensor& input_tensor = context->input(input_index);
    auto input_data = input_tensor.flat<T>();
    int batch_size = input_tensor.dim_size(0);
    int batch_length = input_tensor.NumElements()/batch_size;
    
    Tensor* output_tensor = nullptr;
    TensorShape output_shape = input_tensor.shape();
    output_shape.set_dim(0,sum_rates);
    OP_REQUIRES_OK(context, context->allocate_output(input_index-1, output_shape,&output_tensor));
    auto output_data = output_tensor->flat<T>();
    
    unsigned int output_batch_index = 0;
    for (unsigned int ibatch = 0; ibatch<batch_size; ++ibatch)
    {
        for (unsigned int icopy = 0; icopy<diced_rates[ibatch];++icopy)
        {
            for (unsigned int idata = 0; idata<batch_length; ++idata)
            {
                output_data(output_batch_index*batch_length+idata) = input_data(ibatch*batch_length+idata);
            }
            output_batch_index+=1;
        }
    }
}
